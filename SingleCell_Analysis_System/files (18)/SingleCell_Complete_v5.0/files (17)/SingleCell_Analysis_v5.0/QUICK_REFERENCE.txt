================================================================================
⚙️ 关键参数修改（在Run_Complete_Analysis.R中）
================================================================================

params <- list(
  min_features = 200,      # 细胞最少基因数 [推荐: 200-500]
  max_features = 6000,     # 细胞最多基因数 [推荐: 5000-8000]
  mt_cutoff = 20,          # 线粒体% [正常:15-20, 肿瘤:20-30]
  n_features = 2000,       # 高变基因数 [推荐: 2000-3000]
  n_pcs = 30,              # 主成分数 [推荐: 20-50]
  resolution = 0.8         # 聚类分辨率 [细分:1.0-1.5, 粗分:0.4-0.6]
)

🔧 常用命令

# 加载已保存的对象
pbmc <- readRDS("RData_Storage/Seurat_Object.rds")

# 查看基本信息
ncol(pbmc)                             # 细胞数
nrow(pbmc)                             # 基因数
table(pbmc$cell_type)                  # 细胞类型分布
table(pbmc$seurat_clusters)            # 聚类分布

# 生成自定义图
DimPlot(pbmc, group.by = "cell_type")  # UMAP by cell type
FeaturePlot(pbmc, features = "CD3D")   # 基因表达图
VlnPlot(pbmc, features = "CD3D")       # 小提琴图

# 提取特定细胞类型
tcells <- subset(pbmc, subset = cell_type == "T cells")

# 提取marker基因
markers <- read.csv("RData_Storage/All_Markers.csv")
top_markers <- markers %>% 
  group_by(cluster) %>% 
  top_n(10, avg_log2FC)

# 清除断点重新运行
file.remove(list.files(pattern = "^checkpoint_"))

================================================================================
⚠️ 常见问题速查
================================================================================

问题1: 内存不足
解决: memory.limit(32000)  # Windows
      或减少细胞数测试

问题2: 包安装失败
解决: options(repos = "https://cloud.r-project.org/")
      install.packages("包名", dependencies = TRUE)

问题3: GetAssayData错误
解决: 已在v5.0修复！确保使用最新代码

问题4: 没有生成某些图
原因: 可能缺少分组信息（需要多个样本）
检查: table(pbmc$orig.ident)

问题5: jjVolcano安装失败
解决: install.packages("scRNAtoolVis")
      或跳过高级可视化部分

问题6: 分析中断
解决: 再次运行会自动从断点恢复
      source("Run_Complete_Analysis.R")
